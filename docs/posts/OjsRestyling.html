// --- Reusable Bootstrap Input Group Factory --- //

function InputGroup({ placeholder = "Type…", button = "Go" } = {}) {
  // 1. Create the input
  const input = html`<input type="text" class="form-control" placeholder="${placeholder}">`;

  // 2. Create the button
  const btn = html`<button class="btn btn-primary">${button}</button>`;

  // 3. Create the input-group wrapper
  const group = html`<div class="input-group"></div>`;
  group.append(input, btn);

  // 4. Reactive value
  group.value = "";

  // 5. Emit reactive updates on button click
  btn.addEventListener("click", () => {
    group.value = input.value;
    group.dispatchEvent(new CustomEvent("input"));
  });

  return group;
}
